#Python PyPI installation
```
# Red Hat
sudo yum install -y python-pip
# Debian
sudo apt install -y python-pip
# Linux/Unix
sudo easy_install pip
# Mac OS X
brew install python-pip
# On all systems using pip repo install
sudo pip install ansible
# On all systems using GitHub project
sudo pip install git+https://github.com/ansible/ansible.git@devel
# Ansible Version selection
sudo pip install ansible==2.4.0
# Ansible upgrade
sudo pip install ansible --upgrade
```
